\section{Objectives}

\begin{frame}{Objectives}
	\begin{itemize}
		\item An ABM opinion dynamics model was developed to reproduce the opinion change during the Mexican 2024 presidential election. It was found to be capable of reproducing the main pattern considered: empirical preference share registered in a selected opinion poll.
		
		\item For the current experiments, we follow Pattern Oriented Modeling (POM) \cite{railsbackAgentbasedIndividualbasedModeling2019} and evaluate the model's capability to reproduce multiple patterns observed in an election, such as vote share by gender and age group.
	\end{itemize}
\end{frame}

\begin{frame}{Structural realism}
	\begin{itemize}
		\item An ABM should be as simple as possible while still capturing and reproducing the key characteristics and behaviors of the real system, based on the objectives of the simulation.
		
		\item This is called \emph{structural realism}: including only the essential components of a real system that are needed to produce the main properties and behaviors. 
	\end{itemize}
\end{frame}

\begin{frame}{Pattern Oriented Modeling (POM)}
	\begin{itemize}
		\item A model can be capable of reproducing the main pattern observed in a phenomena while not having structural realism.
		
		\item POM helps to identify the main components necessary in the model, by using \emph{multiple} observed patterns in the real system. A model capable of reproducing multiple patterns is more likely to have achieved structural realism.
	\end{itemize}
\end{frame}

\section{Materials and methods}
\begin{frame}{Chosen opinion polls}
	\begin{itemize}
		\item The proposed model is based on the study carried over by the group GEA-ISA for the election analyzed \cite{gea2024estudios}. It reports the voting intention of 1070 mexican citizens interviewed at a nationwide scale, registering multiple demographic variables. 
		
		\item The data was adjusted to consider only the two main candidates. The election results were also considered, and the resulting preference share is as shown in the following table.

\begin{frame}{Basic model}
	\begin{itemize}
		\item A basic model of opinion dynamics was developed considering the poll's characteristics and the restrictions of a political election. The model considers three different types of influence, based on a classification given by \cite{flacheModelsSocialInfluence2017b, smaldinoModelingSocialBehavior2023}.
		
		\item The model consist of 1070 individual, with every time step representing a day. As such, a run consist of 260 time steps or ticks, considering the days since the first poll until the election day. The initial preference share is taken from the first available poll.
		
		\item The model was capable of reproducing the main pattern of preference share at the three dates with empirical data.
	\end{itemize}	
\end{frame}

\begin{frame}{Limitations of the basic model}
	\begin{itemize}
		\item The developed basic model doesn't consider any characteristic for the agents besides their initial preference. As such, the only pattern capable to be validated is the preference share.
		
		\item To validate multiple patterns, the model needs to be modified so the agents have multiple characteristics beyond their opinion. This is done considering age and gender.
	\end{itemize}

\subsection{Preference share by gender}

\begin{frame}{Obtaining preference share by gender}
	\begin{itemize}
		\item Survey data is weighted to represent the proportions reported in the national electoral list, and this was taken into account while reducing the data to the two main options.
		
		\item The electoral registry shows an electorate composed of $48\%$ men and $52 \%$ women. 
		
		\item The preference share by gender was calculated for each one of the available polls, first dividing the population by gender and then calculating each subgroup preferences.
	\end{itemize}
\end{frame}

\subsection{Validating secondary patterns: preference share by age}
\begin{frame}{Changes to basic model}
	\begin{itemize}
		\item As a first step, the model will be modified to verify the preference share by gender, which will be validated with comparing the simulated results to the reported data.
		\item To validate multiple patterns, the agent's present in the model must be modified to add gender as a variable, taking the gender proportion and preferences from the first poll.
		\item The simulated data will be compared to the second and third poll measuring the RMSE. Given that the election data doesn't report results by gender, they will not be taken into account.
	\end{itemize}
\end{frame}

\begin{frame}[allowframebreaks]{Initialization of gender and preference}
	
	\begin{itemize}
		\item The model is initialized with $48\%$ of male agents and $52\%$ of female agents. following the registered electorate in the same way as the selected surveys. This proportion is kept constant during the time period considered in the model.
		\item Breaking down the population by gender, the preferences for option A and B are taken from table \ref{tab:preferencia_genero}. 
		\item The overall preference for A is $61\%$ and $39\%$ for B at the first time step, the same as the ones registered in the first survey and the ones utilized in the basic model. As such, both can be compared fairly.
	\end{itemize}

\begin{frame}{Interaction between agents}
	\begin{itemize}
		\item The main scheduling from the basic model hasn't been changed. As such, a certain number of pairs of agents are selected to interact every tick at random, not taking their gender into consideration.
		\item Agents interact and change their opinion based on the simulated influence type. After 260 ticks, corresponding to the number of days since the first poll and the election, the model stops and the relevant data is captured.
	\end{itemize}
\end{frame}

\section{Methodology}
\begin{frame}{Comparison by type of influence}
	\begin{itemize}
		\item Taking the results from the basic model, the configuration with the lowest RMSE for each type of influence was selected. Each configuration was executed for 30 repetitions, and the general preference and preference by gender was recorded at each time step.
		
		\centering
		\begin{table}
			\resizebox{0.95 \textwidth}{!}{
				\begin{tabular}{|c|c|c|c|}
					\hline
					& \textbf{Positive influence} & \textbf{Negative influence} & \textbf{Bounded confidence} \\
					\hline
					\textit{agents per tick} & 20 & 20 & 30 \\
					\hline
					\textit{learning rate} & 0.1 & 0.1 & 0.1 \\
					\hline
					\textit{confidence threshold} & - & 1 & 0.8 \\
					\hline
					\textit{RMSE} & $2.56 \pm 0.36$ & $2.4 \pm 0.23$ & $2.3 \pm 0.32$ \\
					\hline
				\end{tabular}
			}
		\caption{Configuration with the lowest RMSE by type of influence. Results from previous experiments.}
		\end{table}
	\end{itemize}
\end{frame}

\begin{frame}{Measuring the error}
	\begin{itemize}
		\item The RMSE for general preference was measured, as well as the RMSE of preference by gender compared to the registered data. This is done taking into consideration only option A. Given that the values are reciprocal, the RMSE for option B will have the same value.
		\item The preference share by gender is only measured compared to the second and third poll, not taking into account the final vote.
		\item For the general preference we expect an RMSE similar to the one obtained in previous experiments, and a poor performance when sectioning the population by gender.
	\end{itemize}
\end{frame}

\section{Conclusions}
\begin{frame}{Conclusions}
	\begin{itemize}
		\item The gender of the agents plays no part in the mechanism for selecting a partner to interact with. As expected, when the population is divided by gender and the preference evolution is calculated, it repeats the main pattern observed over the general population. This holds true for the three types of influence.
		\item The mean RMSE is similar for the three patterns, and the model is incapable of reproducing more complex preference changes as the ones observed in the empirical male preference.
		\item A more robust model must incorporate the agent's characteristics for the interaction mechanism, generating a more complex social network in the agents present in the model.
	\end{itemize}
\end{frame}
